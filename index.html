        const giorni = ['lun', 'mar', 'mer', 'gio', 'ven', 'sab'];

        const giorniNome = {

          'lun': 'Lunedì',

          'mar': 'Martedì',

          'mer': 'Mercoledì',

          'gio': 'Giovedì',

          'ven': 'Venerdì',

          'sab': 'Sabato'

        };

        document.write(giorni.map(g =>

          `<tr>

            <td>${giorniNome[g]}</td>

            <td><input type="checkbox" name="${g}_mattina" value="x" data-giorno="${g}" data-fascia="Mattina"></td>

            <td><input type="checkbox" name="${g}_pomeriggio" value="x" data-giorno="${g}" data-fascia="Pomeriggio"></td>

  <div id="riepilogo">

    <h3>Riepilogo Turni Selezionati</h3>

    <ul id="listaRiepilogo"></ul>

  </div>


  <script>

    const riepilogo = document.getElementById('listaRiepilogo');

    const checkboxes = document.querySelectorAll('input[type="checkbox"]');


    function aggiornaRiepilogo() {

      riepilogo.innerHTML = '';

      checkboxes.forEach(cb => {

        if (cb.checked) {

          const giorno = cb.dataset.giorno;

          const fascia = cb.dataset.fascia;

          const nomeGiorno = giorniNome[giorno];

          const id = `${giorno}_${fascia}`;


          const li = document.createElement('li');

          li.textContent = `${nomeGiorno} - ${fascia}`;

          li.id = 'riepilogo_' + id;


          const btn = document.createElement('span');

          btn.textContent = '❌';

          btn.className = 'remove-btn';

          btn.onclick = () => {

            cb.checked = false;

            aggiornaRiepilogo();

          };


          li.appendChild(btn);

          riepilogo.appendChild(li);

        }

      });

    }


    checkboxes.forEach(cb => {

      cb.addEventListener('change', aggiornaRiepilogo);

    });


    aggiornaRiepilogo();

  </script>
